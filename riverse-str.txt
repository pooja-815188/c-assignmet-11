#include<stdio.h>
#include<string.h>
int main()
{
    char str[20],rev[20];
    int i,j,count=0;
    printf("enter a string\n");
    scanf("%s",str);
    int temp;
    int l=strlen(str);
    j=l-1;
    for(i=0,j;i<l;i++,j--)
    {
        temp=rev[i];
        rev[i]=str[j];
        str[j]=temp;
    }
    printf("reverse of string is %s\n",rev);
}
